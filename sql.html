Databricks SQL Basics

Creating & Using Databases

-- Create a database if it does not exist
CREATE DATABASE IF NOT EXISTS my_database;
USE my_database;

Creating Tables

-- Create a managed Delta table
CREATE TABLE sales (
    order_id INT,
    customer STRING,
    amount DOUBLE,
    order_date DATE
) USING DELTA;

Inserting Data

-- Insert data into the table
INSERT INTO sales VALUES (1, 'Alice', 100.50, '2024-01-01');
INSERT INTO sales VALUES (2, 'Bob', 200.75, '2024-01-02');

Querying Data

-- Retrieve all records
SELECT * FROM sales;

-- Filtered query
SELECT * FROM sales WHERE amount > 100;

Updating & Deleting Data

-- Update a record
UPDATE sales SET amount = 250 WHERE order_id = 1;

-- Delete a record
DELETE FROM sales WHERE order_id = 2;

Exploring Table Data

-- Describe table schema
DESCRIBE TABLE sales;

-- Show detailed table info
DESCRIBE DETAIL sales;
